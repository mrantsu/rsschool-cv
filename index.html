function smallEnough(a, limit) {
  return a.every(elem => elem <= limit);
}

                        

function isIsogram(str){
  return new Set(
  str.toLowerCase().split('')).size === str.length
  );
}

                        

function deepCount(a) {
  let result = a.length;
  for(let item of a) {
    if(Array.isArray(item)){
      result += deepCount(item);
    }
  }
  return result;
}

                        

const validWord = function(dictionary, word){
  const pearseDictionary = dictionary.join('|');
  const regExp = new RegExp(`^(${pearseDictionary})+$`);
  return regExp.test(word);
};

                        

function zipWith(fn,a0,a1) {
  const result = [];
  let length = (a0.length > a1.length) ? a1.length :   a0.length;
  for(let i = 0; i < length; i++){
    result.push(fn(a0[i],a1[i]));
  }
  return result;
};

                        

function nthFibo(n) {
  let solution;
  if(n <= 2) return n - 1;
  solution = nthFibo(n - 1) + nthFibo(n - 2);
  return solution;
}
